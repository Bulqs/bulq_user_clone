          {/* Render inner midek on top of the first model for second div */}
          {isInnerModalOpen2 && (
            <div className="fixed inset-0 pt-2 md:pt-4 flex items-start justify-center bg-appBlack/70 z-60">
              <div className="bg-appTitleBgColor px-4 pt-2 md:pt-4 md:px-4 p-2 md:py-4 rounded-lg shadow-lg flex flex-col items-center justify-center">
                <BannerStepForm2 /> {/* Place the form here */}
                <button onClick={() => setIsInnerModalOpen2(false)} className="lg:mt-2 px-4 py-2 bg-appBanner text-appWhite rounded lg:ml-4 flex items-center justify-between gap-2 ">
                  <IoMdCloseCircle className="text-appWhite text-2xl" />  Close Form
                </button>
              </div>

            </div>
          )}

        </div>
      )}
      {/* second modal section ends here */}


      {isModalOpen3 && selectedCard3 && (
        <div className="fixed inset-0 pt-2 md:pt-4 flex items-start justify-center bg-appBlack/70 z-50">
          <div className="bg-appTitleBgColor px-4 pt-2 md:pt-4 md:px-4 p-2 md:py-4 rounded-lg shadow-lg flex flex-col items-center justify-center">
            {/* <h3 className="text-xl font-bold">{selectedCard3.title}</h3>
            <p>{selectedCard3.description}</p> */}
            <BannerStepForm3 />
            <button onClick={closeModal3} className="lg:mt-2 px-4 py-2 bg-appBanner text-appWhite rounded lg:ml-4 flex items-center justify-between gap-2 ">
              <IoMdCloseCircle className="text-appWhite text-2xl" />  Close Form
            </button>
          </div>
        </div>
      )}

    </div>
  )
}
